<script>
$(document).ready(function(){
var activeFloatAd = function(dom,type) {
var type = type ? type : 'top';
var ad = scrollMonitor.create(dom);
var footer = scrollMonitor.create($('.foot'),{
top: (type == 'top') ? parseInt(dom.css('height')) : 0
});
var fixedClass = 'fixed-ads sl-fixed sl-fixed-' + type + ' fixed-ads-offset-' + type;
var absoluteClass = 'absolute-ads';
ad.lock();
if (type == 'top') {
ad.stateChange(function() {
if (dom.hasClass(absoluteClass)) {
dom.removeClass(absoluteClass);
};
dom.toggleClass(fixedClass, this.isAboveViewport);
});
ad.exitViewport(function(){
if (ad.isAboveViewport && dom.hasClass('fixed-ads') == false) {
dom.addClass(fixedClass);
}
})
footer.fullyEnterViewport(function(){
if (this.isAboveViewport) {
dom.removeClass(fixedClass).addClass(absoluteClass);
}
});
footer.partiallyExitViewport(function(){
if (this.isBelowViewport) {
dom.removeClass(absoluteClass).addClass(fixedClass)
}
});
} else {
ad.stateChange(function() {
if (!dom.hasClass(absoluteClass)) {
dom.toggleClass(fixedClass, !this.isFullyInViewport && this.isAboveViewport);
}
});
ad.partiallyExitViewport(function(){
if (ad.isAboveViewport && !ad.isFullyInViewport && dom.hasClass('fixed-ads') == false && !dom.hasClass(absoluteClass)) {
dom.addClass(fixedClass);
}
})
footer.enterViewport(function(){
dom.removeClass(fixedClass).addClass(absoluteClass);
});
footer.exitViewport(function(){
if (this.isBelowViewport) {
dom.removeClass(absoluteClass).addClass(fixedClass)
}
});
}
return {
ad: ad,
footer: footer
};
}
